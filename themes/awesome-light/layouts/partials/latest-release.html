{{/* Fetches the latest GitHub release (including prereleases) for a repo URL and returns "vX.Y.Z|2026-07-20|https://github.com/owner/name/releases/tag/vX.Y.Z", or "" if the repo has no release / isn't on GitHub / site is running locally. Expects: . = repo URL (e.g. "https://github.com/owner/name") */}} {{ $repo := . }} {{ $out := "" }} {{ if and (hasPrefix $repo "https://github.com/") (not hugo.IsServer) }} {{ $slug := strings.TrimPrefix "https://github.com/" $repo }} {{ $slug = strings.TrimSuffix "/" $slug }} {{ $api := printf "https://api.github.com/repos/%s/releases?per_page=1" $slug }} {{ $opts := dict "headers" (dict "Accept" "application/vnd.github+json") }} {{ with os.Getenv "GITHUB_TOKEN" }} {{ $opts = dict "headers" (dict "Accept" "application/vnd.github+json" "Authorization" (printf "Bearer %s" .)) }} {{ end }} {{ $try := try (resources.GetRemote $api $opts) }} {{ with $try.Value }} {{ $releases := . | transform.Unmarshal }} {{ if gt (len $releases) 0 }} {{ $data := index $releases 0 }} {{ with $data.tag_name }} {{ $date := "" }} {{ with $data.published_at }}{{ $date = (time.AsTime .).Format "2006-01-02" }}{{ end }} {{ $out = printf "%s|%s|%s" . $date ($data.html_url | default "") }} {{ end }} {{ end }} {{ end }} {{ end }} {{ return $out }}
